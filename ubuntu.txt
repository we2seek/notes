# Сжать папку и разбить архив через 7z
	7z a -tzip -v1m -mx9 result.zip source_folder/


# Как разбить tar.zg архив на тома и как его потом склеить. Рекомендуется использовать tar, а не 7z, который проще, т.к. 7z не сохраняет права доступа к файлам

# Сжать и разбить:
	tar czf - ./backup | split -d -b 10m - backup.tar.gz.

# Склеить и распаковать:
	cat backup.tar* | tar xzf -
